
SET SQL_SAFE_UPDATES = 0;
update loans l join customers c on l.CustomerID=c.CustomerID 
set l.InterestRate=(0.99*l.InterestRate) where timestampdiff(year,c.dob,curdate())>60; 
SET SQL_SAFE_UPDATES = 1;
-- Scenario 1

SELECT CONCAT(
    'Reminder: Dear ',
    CustomerName,
    ', your loan is due on ',
    DueDate
) AS Message
FROM Loans
WHERE DueDate BETWEEN CURDATE()
                 AND DATE_ADD(CURDATE(), INTERVAL 30 DAY);
